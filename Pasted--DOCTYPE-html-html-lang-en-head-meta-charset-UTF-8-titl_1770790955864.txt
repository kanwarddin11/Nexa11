            <div class="grid grid-cols-1 md:grid-cols-3 gap-6 max-w-5xl mx-auto">

    <script>
        async function processData() {
            const input = document.getElementById('userInput').value;
            const resBox = document.getElementById('resultBox');
            
            const response = await fetch('/api/verify', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({input_data: input})
            });
            const data = await response.json();
            
            document.getElementById('score').innerText = data.trust_score + "%";
            document.getElementById('verdict').innerText = data.verdict;
            document.getElementById('origin').innerText = data.origin_source;
            resBox.classList.remove('hidden');
        }
    </script>
